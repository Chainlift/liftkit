## Base Component

**Default Behavior**

A simple `<i>` element with a single slot for accepting an icon. LiftKit uses Material Symbols for simplicity and consistency across platforms.

### HTML

```html
<i lk-component="icon">add</i>

```

### TypeScript

```tsx
import { useMemo } from 'react';
import { propsToDataAttrs } from './utilities';

interface LkIconProps extends React.HTMLAttributes<HTMLElement> {
  fontClass?: Exclude<LkFontClass, `${string}-bold` | `${string}-mono`>;
  color?: keyof LkColor | 'currentColor';
  display?: 'block' | 'inline-block' | 'inline';
  children: string; // material symbol name
}

export default function Icon(props: LkIconProps) {
  const { children, ...restProps } = props;

  const lkIconAttrs = useMemo(
    () => propsToDataAttrs(restProps, 'lk-icon'),
    [restProps.fontClass, restProps.color, restProps.display]
  );

  return (
    <i {...lkIconAttrs}>{children}</i>
  );
}

```

### CSS

```css
[lk-component="icon"] {
  font-family: 'Material Symbols Outlined';
  font-weight: normal;
  font-style: normal;
  font-size: var(--lk-size-md);
  line-height: 1;
  letter-spacing: normal;
  text-transform: none;
  display: block;
  white-space: nowrap;
  direction: ltr;
  -webkit-font-feature-settings: 'liga';
  -webkit-font-smoothing: antialiased;
}

```

## Combine with Native HTML Attributes

You can set any other *native* HTML attribute (className, style, id, etc.) on the component by passing it as a prop.

## Props

**fontClass**

**Description**

Controls the size of the icon via typography class.

| **Expected Type** | **Default Value** | **Acceptable Values** |
| --- | --- | --- |
| `LkFontClass` (excluding `-bold`, `-mono`) | `body` | Any non-suffixed LkFontClass value |

```tsx
<Icon fontClass="title2">add</Icon>

```

**color**

**Description**

Controls the color of the icon.

| **Expected Type** | **Default Value** | **Acceptable Values** |
| --- | --- | --- |
| `keyof LkColor` or `'currentColor'` | `currentColor` | Any LkColor key or 'currentColor' |

```tsx
<Icon color="lk-primary">star</Icon>

```

**display**

**Description**

Controls the CSS `display` property of the icon.

| **Expected Type** | **Default Value** | **Acceptable Values** |
| --- | --- | --- |
| `string` | `block` | `block`, `inline-block`, `inline` |

```tsx
<Icon display="inline">close</Icon>

```
