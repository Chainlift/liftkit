# Dropdown Chip

## Base Component

**Dropdown Chip**

A `<select>`-like control that visually resembles a text input. It uses the same CSS properties as the TextInput component, including layout, spacing, border behavior, and interactive states. By default, the chip includes a downward arrow icon (`keyboard_arrow_down`) inside a special container with the attribute `lk-dropdown-el="chip"`.

### HTML

```
<div lk-component="dropdown-chip" lk-dropdown-chip-is-active="false">
  <label lk-text-input-element="label">Select an option</label>
  <div lk-text-input-element="field-wrap">
    <input type="text" readonly value="Option 1" lk-text-input-element="input" />
    <div lk-component="icon" lk-text-input-element="end-icon">keyboard_arrow_down</div>
    <div lk-component="state-layer"></div>
  </div>
</div>
```

### TypeScript

```
import { useMemo } from 'react';
import {propsToDataAttrs} from "../utilities"

interface LkDropdownChipProps extends React.InputHTMLAttributes<HTMLInputElement> {
  labelPosition?: 'default' | 'on-input';
  helpText?: boolean;
  placeholderText?: string;
  isActive?: boolean;
}

export default function DropdownChip(props: LkDropdownChipProps) {
  const { children, ...restProps } = props;

  const dropdownChipAttrs = useMemo(
    () => propsToDataAttrs(restProps, 'lk-dropdown-chip'),
    [restProps.labelPosition, restProps.helpText, restProps.placeholderText, restProps.isActive]
  );

  return (
    <div {...dropdownChipAttrs}>{children}</div>
  );
}
```

### CSS (Reference)

```
[lk-component="dropdown-chip"] {
  display: block;
}

[lk-component="dropdown-chip"] [lk-component="state-layer"] {
  transition: background-color 0.2s ease-out, opacity 0.2s ease-out;
  background-color: var(--lk-primary);
  opacity: 0;
}

[lk-dropdown-chip-is-active="true"] [lk-component="state-layer"] {
  opacity: 0.15;
}
```

## Combine with Native HTML Attributes

You can set any other _native_ HTML attribute (className, style, id, etc.) on the component by passing it as a prop.

## Props

### **isActive**

**Description**

Controls whether the dropdown list is visible. This value is passed down to the `DropdownList` component and used to animate the state-layer and other visibility toggles. When active, the chip displays its state-layer at 15% opacity.

| **Expected Type** | **Default Value** | **Acceptable Values** |
| ----------------- | ----------------- | --------------------- |
| `boolean`         | `false`           | `true`, `false`       |

```
<DropdownChip isActive={true} />
```

All other props are inherited from the `TextInput` component: `labelPosition`, `helpText`, and `placeholderText`.
