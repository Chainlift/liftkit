# Row

## Base Component

### HTML

```html
html
CopyEdit
<div lk-component="row"
     lk-row-align-items="start"
     lk-row-justify-content="start"
     lk-row-wrap-children="false"
     lk-row-default-child-behavior="ignoreFlexRules"
     lk-row-gap="none">
  {children}
</div>
```

### TypeScript

```tsx
tsx
CopyEdit
import { useMemo } from 'react';
import { propsToDataAttrs } from './utilities';

type LkSizeUnit =
  | '3xs' | '2xs' | 'xs' | 'sm' | 'md'
  | 'lg' | 'xl' | '2xl' | '3xl' | '4xl';

interface LkRowProps extends React.HTMLAttributes<HTMLDivElement> {
  alignItems?: 'start' | 'center' | 'end' | 'stretch';
  justifyContent?: 'start' | 'center' | 'end' | 'space-between' | 'space-around';
  gap?: LkSizeUnit;
  wrapChildren?: boolean;
  defaultChildBehavior?: 'auto-grow' | 'auto-shrink' | 'ignoreFlexRules' | 'ignoreIntrinsicSize';
}

export default function Row(props: LkRowProps) {
  const { children, ...restProps } = props;

  const lkRowAttrs = useMemo(
    () => propsToDataAttrs(restProps, 'lk-row'),
    [restProps.alignItems, restProps.justifyContent, restProps.gap, restProps.wrapChildren, restProps.defaultChildBehavior]
  );

  return (
    <div {...lkRowAttrs}>
      {children}
    </div>
  );
}

```

### Combine with Native HTML Attributes

You can set any native HTML attributes (like `className` or `style`) on the component by passing them directly as props:

```tsx

<Row className="my-custom-row" style={{ margin: '2rem' }} />
```

---

### Props

### `alignItems`

### Description

Modifies the `align-items` CSS property. Controls how elements align with respect to the y-axis.

| **Expected Type** | **Default Value** | **Acceptable Values** |
| --- | --- | --- |
| `string` | `"start"` | `"start"`, `"center"`, `"end"`, `"stretch"` |

```tsx
<Row alignItems="center" />
```

---

### `justifyContent`

### Description

Modifies the `justify-content` CSS property. Controls how flex children are distributed.

| **Expected Type** | **Default Value** | **Acceptable Values** |
| --- | --- | --- |
| `string` | `"start"` | `"start"`, `"center"`, `"end"`, `"space-between"`, `"space-around"` |

```tsx
<Row justifyContent="space-between" />
```

---

### `gap`

### Description

Modifies the `gap` between children.

| **Expected Type** | **Default Value** | **Acceptable Values** |
| --- | --- | --- |
| `LkSizeUnit` | `"none"` | `'3xs'`, `'2xs'`, `'xs'`, `'sm'`, `'md'`, `'lg'`, `'xl'`, `'2xl'`, `'3xl'`, `'4xl'` |

```tsx
<Row gap="lg" />
```

---

### `wrapChildren`

### Description

Controls the `flex-wrap` property. If true, the children wrap to the next line.

| **Expected Type** | **Default Value** | **Acceptable Values** |
| --- | --- | --- |
| `boolean` | `false` | `true`, `false` |

```tsx
<Row wrapChildren={true} />
```

---

### `defaultChildBehavior`

### Description

Controls all children's flex behavior. For simple layout use cases, lets you apply uniform flex settings to all children.

| **Expected Type** | **Default Value** | **Acceptable Values** |
| --- | --- | --- |
| `string` | *(none)* | `"auto-grow"`, `"auto-shrink"`, `"ignoreFlexRules"`, `"ignoreIntrinsicSize"` |

```tsx
<Row defaultChildBehavior="auto-grow" />
```
